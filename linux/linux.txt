#Выведите список всех файлов в папке html

cd var/www/html /Перехожу в папку html
ls -a	/Вывожу содержимое папки html, опция -a выводит скрытые файлы

#Найдите в файле access.log все упоминания строки "404"
cd var/logs /Перехожу в папку logs
grep 404 access.log /Вывожу упоминания 404 в logs

#На основании файла 404.html сделайте файл 403.html заменив в его содержимом все упоминания 404 на 403
cd var/www/html /Перехожу в папку html
cp 404.html 403.html /Копируем файл и меняем имя на 403
sed -i 's/404/403/d' 403.html /Заменяем упоминание 404 на 403 в файле403.html

#Модифицируйте права доступа и владельца файла index.html следующим образом: Текущий пользователь должен иметь права на чтение/запись Группа www-data на чтение Все остальные не имеют доступа к файлу
cd var/www/html /Перехожу в папку html
chgrp www-data index.html /Изменяем группу для файла
chmod u=rw index.html /Изменяем для юзера право доступа на чтение и изменение
chmod g=r index.html /Изменяем для группы право доступа на чтение
chmod o-rwx index.html /Запрещаем читать, изменять и использовать файл для остальных пользователей

#Сделайте символическую ссылку на файл access.log в папке html.
cd var /Перехожу в папку var
ls -rs log/access.log /www/html/access.log /Создаю и помещаю динамичную символическую ссылку из дериктории log/access.log в дерикторию www/html/access.log

#Выведите на экран все файлы в папке var
ls -aR var /Рекрусивно вывожу все файлы в папке var
